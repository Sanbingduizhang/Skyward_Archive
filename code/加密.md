##### hash加密规则一(go)
```golang
query := fmt.Sprintf("app=%s&key=%s&ts=%s", app, key, ts)  
  
signBase := r.Method + r.Host + r.URL.Path + "?" + query  
  
// ts+app+key -> SHA256 -> hex  
shaPart := fmt.Sprintf("%x", sha256.Sum256([]byte(ts+app+key)))  
  
// 拼接最终串  
tmpEncode := signBase + shaPart  
  
hmacValue := hmacSHA256(tmpEncode, secretKey)  
expectSign := base64.StdEncoding.EncodeToString(hmacValue)  
  
if !hmac.Equal([]byte(sign), []byte(expectSign)) {  
    return fmt.Errorf("sign mismatch")  
}
```
